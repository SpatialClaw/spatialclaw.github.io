def normalize(v):
    norm = np.linalg.norm(v)
    return v / norm if norm > 0 else v

def simulate_path(turn1_type, turn2_type):
    curr_p = p_door.copy()
    curr_v = v_init.copy()

    if turn1_type == 'right':
